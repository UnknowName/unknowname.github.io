将本地系统的/etc/yum.repos.d/目录中的yum源同步至本地，这样本地局域网就可以使用内网的yum源，加快速度

安装工具

1

yum install nginx yum-utils -y

#!/bin/bash
export PATH

cd /data/yumdata
reposync -nd
dirs=$(ls -F |grep "/")
for d in ${dirs};
do
 cd /data/yumdata/${d}
 createrepo --update ./
done
